Unique Color Shirt

      function Unique_Shirts (arr,N) {

        var unique=0;
        var secondarr=[];
        var count=0;
        for(i=0;i< N;i++){
          
          for(j=0;j< N;j++){
           if(i!=j){
             if(arr[i]==arr[j]){
               count++;
             }
           }
          }
          secondarr.push(count);
          count=0
        }
        for(i=0;i< secondarr.length;i++){
          if(secondarr[i]===0){
            unique++
          }
        }
      
      return   unique;
      }

        function arrayMin(arr) {
   
            var min=100;
           for(i=0;i<=arr.length;i++){
             if (arr[i]< min){

           }  
           return min
         }
         
         function arrayMax(arr) {
